     7→export interface SkillNode {
     8→  id: string;
     9→  title: string;
    10→  branch: string;
    11→  tier: number; // tier within branch (1 = foundation, higher = advanced)
    12→  xp: number; // XP awarded for mastery
    13→  marks: string; // e.g. "5m", "12m", "2m"
    14→  description: string;
    15→  highYield: 1 | 2 | 3 | 4 | 5; // 5 = appears most frequently in past papers
    16→  prerequisites: string[]; // ids of nodes that must be mastered first
    17→  keyPoints: string[];
    18→  textbookRef: string;
    19→}
    20→
    21→export interface SkillBranch {

    23→  name: string;

    25→  color: string; // hex color for the branch theme

    27→  description: string;
    28→  nodes: SkillNode[];
    29→}
